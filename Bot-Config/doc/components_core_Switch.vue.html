<template>
  <md-layout md-align="end">
    <label class="switch">
      <input type="checkbox" v-model="checked" @click="handleClick">
      <span class="slider round" :class="theme"></span>
    </label>
  </md-layout>
</template>

<script>
/**
 * A switch component.
 *
 * @author Marcel Herd
 * @module components/core/Switch
 *
 * @param {boolean} initial - intial value of the switch
 * @param {string} [theme=default] - default|white, use white theme on white background
 */
export default {
  name: 'bt-switch',
  props: ['initial', 'theme'],
  computed: {
    checked () {
      return this.initial
    }
  },
  methods: {
    /**
     * Propagates click event.
     *
     * @method handleClick
     */
    handleClick () {
      this.$emit('click')
    }
  }
}
</script>
